### 1. NTP 网络时间校验 ⭐⭐⭐⭐⭐

**问题**：原方案使用 `DateTime.Now`，孩子可修改系统时间绕过限制

**解决方案**：从网络时间服务器验证时间

```csharp
using System;
using System.Net;
using System.Net.Sockets;
using System.Threading.Tasks;

namespace ChildControl.Protection
{
    public class NtpTimeValidator
    {
        private static readonly string[] NtpServers = 
        {
            "pool.ntp.org",
            "time.windows.com",
            "time.google.com",
            "cn.pool.ntp.org"
        };
        
        private const int NtpPort = 123;
        private static DateTime? _cachedNetworkTime;
        private static DateTime _lastCacheTime;
        private static readonly object _lock = new object();
        
        /// <summary>
        /// 获取网络时间（带缓存）
        /// </summary>
        public static DateTime GetNetworkTime(bool forceRefresh = false)
        {
            lock (_lock)
            {
                // 缓存5分钟有效期
                if (!forceRefresh && _cachedNetworkTime.HasValue && 
                    (DateTime.UtcNow - _lastCacheTime).TotalMinutes < 5)
                {
                    return _cachedNetworkTime.Value;
                }
                
                var time = GetNetworkTimeInternal();
                if (time != DateTime.MinValue)
                {
                    _cachedNetworkTime = time;
                    _lastCacheTime = DateTime.UtcNow;
                    return time;
                }
                
                // 如果获取失败，返回当前时间
                return DateTime.UtcNow;
            }
        }
        
        /// <summary>
        /// 从多个NTP服务器获取时间
        /// </summary>
        private static DateTime GetNetworkTimeInternal()
        {
            foreach (var server in NtpServers)
            {
                try
                {
                    var time = GetNtpTime(server);
                    if (time != DateTime.MinValue)
                    {
                        return time;
                    }
                }
                catch { }
            }
            return DateTime.MinValue;
        }
        
        /// <summary>
        /// NTP协议实现
        /// </summary>
        private static DateTime GetNtpTime(string server)
        {
            try
            {
                var ntpData = new byte[48];
                ntpData[0] = 0x1B; // LI=0, VN=3, Mode=3 (Client)
                
                using (var client = new UdpClient())
                {
                    client.Connect(server, NtpPort);
                    client.Send(ntpData, ntpData.Length);
                    
                    client.Client.ReceiveTimeout = 3000;
                    var receiveBytes = client.Receive(ref _);
                    
                    if (receiveBytes.Length == 48)
                    {
                        ulong intPart = BitConverter.ToUInt32(receiveBytes, 40);
                        ulong fracPart = BitConverter.ToUInt32(receiveBytes, 44);
                        
                        intPart = SwapEndianness(intPart);
                        fracPart = SwapEndianness(fracPart);
                        
                        var milliseconds = (intPart * 1000) + ((fracPart * 1000) / 0x100000000L);
                        return new DateTime(1900, 1, 1).AddMilliseconds(milliseconds);
                    }
                }
            }
            catch
            {
                return DateTime.MinValue;
            }
            
            IPEndPoint _ = new IPEndPoint(IPAddress.Any, 0);
            return DateTime.MinValue;
        }
        
        private static uint SwapEndianness(ulong x)
        {
            return (uint)(((x & 0x000000ff) << 24) +
                          ((x & 0x0000ff00) << 8) +
                          ((x & 0x00ff0000) >> 8) +
                          ((x & 0xff000000) >> 24));
        }
        
        /// <summary>
        /// 检测时间是否被篡改
        /// </summary>
        public static bool DetectTimeTampering(TimeSpan tolerance = default)
        {
            if (tolerance == default)
            {
                tolerance = TimeSpan.FromMinutes(5);
            }
            
            var networkTime = GetNetworkTime();
            var systemTime = DateTime.UtcNow;
            var diff = Math.Abs((networkTime - systemTime).TotalMinutes);
            
            bool tampered = diff > tolerance.TotalMinutes;
            
            if (tampered)
            {
                LogTimeTampering(systemTime, networkTime, diff);
            }
            
            return tampered;
        }
        
        /// <summary>
        /// 记录时间篡改行为
        /// </summary>
        private static void LogTimeTampering(DateTime systemTime, DateTime networkTime, double diffMinutes)
        {
            try
            {
                string logEntry = $"[{DateTime.Now:yyyy-MM-dd HH:mm:ss}] TIME TAMPERING DETECTED: " +
                    $"System={systemTime:yyyy-MM-dd HH:mm:ss}, Network={networkTime:yyyy-MM-dd HH:mm:ss}, " +
                    $"Diff={diffMinutes:F2} minutes";
                
                System.IO.File.AppendAllText(
                    @"C:\ProgramData\ChildControl\logs\security.log",
                    logEntry + Environment.NewLine
                );
                
                System.Diagnostics.EventLog.WriteEntry(
                    "ChildControl",
                    $"System time tampered! Diff: {diffMinutes:F2} minutes",
                    System.Diagnostics.EventLogEntryType.Warning
                );
            }
            catch { }
        }
        
        /// <summary>
        /// 获取正确的当前时间（优先网络时间）
        /// </summary>
        public static DateTime GetCurrentTime()
        {
            return GetNetworkTime();
        }
    }
}
```

```csharp
                    // 由于实现复杂，建议使用进程名伪装代替
```

### 3. 进程句柄权限控制 ⭐⭐⭐⭐⭐

**问题**：即使设置 DACL，管理员仍可使用特定工具获取 PROCESS_TERMINATE 权限

**解决方案**：移除进程的 PROCESS_TERMINATE 权限

```csharp
using System;
using System.Diagnostics;
using System.Runtime.InteropServices;
using System.Security.AccessControl;
using System.Security.Principal;

namespace ChildControl.Protection
{
    public class ProcessAccessControl
    {
        [DllImport("kernel32.dll", SetLastError = true)]
        private static extern IntPtr GetCurrentProcess();
        
        [DllImport("kernel32.dll", SetLastError = true)]
        private static extern bool SetKernelObjectSecurity(
            IntPtr Handle,
            int securityInformation,
            byte[] pSecurityDescriptor);
        
        private const int DACL_SECURITY_INFORMATION = 0x00000004;
        
        /// <summary>
        /// 移除进程的终止权限，防止任务管理器Kill
        /// </summary>
        public static bool RemoveTerminatePermission()
        {
            try
            {
                var processHandle = GetCurrentProcess();
                if (processHandle == IntPtr.Zero)
                    return false;
                
                // 创建拒绝终止的 DACL
                var securityDescriptor = new RawSecurityDescriptor("O:BAG:DU");
                var dacl = new DiscretionaryAcl(false, false, 256);
                
                // 拒绝所有人的 PROCESS_TERMINATE 权限
                var everyone = new SecurityIdentifier(WellKnownSidType.WorldSid, null);
                dacl.AddAccess(AccessControlType.Deny, 
                    0x00000001, // PROCESS_TERMINATE
                    0, 
                    everyone, 
                    InheritanceFlags.None, 
                    PropagationFlags.None);
                
                securityDescriptor.DiscretionaryAcl = dacl;
                
                byte[] sdBytes = new byte[securityDescriptor.BinaryLength];
                securityDescriptor.GetBinaryForm(sdBytes, 0);
                
                return SetKernelObjectSecurity(processHandle, DACL_SECURITY_INFORMATION, sdBytes);
            }
            catch (Exception ex)
            {
                EventLog.WriteEntry($"ProcessAccessControl error: {ex.Message}", 
                    System.Diagnostics.EventLogEntryType.Error);
                return false;
            }
        }
        
        /// <summary>
        /// 保护Agent进程
        /// </summary>
        public static void ProtectAgentProcess(int processId)
        {
            try
            {
                var process = Process.GetProcessById(processId);
                var handle = process.Handle;
                
                // 使用低级API移除终止权限
                RemoveTerminatePermission();
            }
            catch { }
        }
    }
}
```

### 4. 虚拟机检测 ⭐⭐⭐

**问题**：孩子可能用虚拟机玩游戏绕过限制

**解决方案**：检测虚拟机特征

```csharp
using System;
using System.Diagnostics;
using System.IO;
using Microsoft.Win32;

namespace ChildControl.Protection
{
    public class VirtualMachineDetector
    {
        /// <summary>
        /// 检测是否在虚拟机中运行
        /// </summary>
        public static bool IsRunningInVM()
        {
            return CheckCpuInfo() ||
                   CheckDiskInfo() ||
                   CheckMacAddress() ||
                   CheckSystemInfo() ||
                   CheckServices();
        }
        
        private static bool CheckCpuInfo()
        {
            try
            {
                var cpuInfo = File.ReadAllText(@"\\.\PhysicalMedia0");
                return cpuInfo.Contains("VMware") ||
                       cpuInfo.Contains("Virtual") ||
                       cpuInfo.Contains("QEMU") ||
                       cpuInfo.Contains("Xen");
            }
            catch { return false; }
        }
        
        private static bool CheckDiskInfo()
        {
            try
            {
                var drives = DriveInfo.GetDrives();
                foreach (var drive in drives)
                {
                    if (drive.VolumeLabel.Contains("VMware") ||
                        drive.VolumeLabel.Contains("Virtual"))
                        return true;
                }
            }
            catch { }
            return false;
        }
        
        private static bool CheckMacAddress()
        {
            try
            {
                var nicCount = NetworkInterface.GetAllNetworkInterfaces().Length;
                return nicCount > 4; // 虚拟机通常有多个虚拟网卡
            }
            catch { return false; }
        }
        
        private static bool CheckSystemInfo()
        {
            try
            {
                var bios = Registry.LocalMachine.OpenSubKey(@"HARDWARE\DESCRIPTION\System");
                if (bios != null)
                {
                    var systemManufacturer = bios.GetValue("SystemManufacturer")?.ToString();
                    return systemManufacturer?.Contains("VMware") == true ||
                           systemManufacturer?.Contains("VirtualBox") == true ||
                           systemManufacturer?.Contains("Xen") == true;
                }
            }
            catch { }
            return false;
        }
        
        private static bool CheckServices()
        {
            try
            {
                var services = ServiceController.GetServices();
                foreach (var service in services)
                {
                    if (service.ServiceName.Contains("VMware") ||
                        service.ServiceName.Contains("VBox") ||
                        service.ServiceName.Contains("Xen"))
                        return true;
                }
            }
            catch { }
            return false;
        }
        
        /// <summary>
        /// 在虚拟机中运行时采取的措施
        /// </summary>
        public static void HandleVirtualMachineEnvironment()
        {
            if (IsRunningInVM())
            {
                EventLog.WriteEntry("Running in virtual machine detected!", 
                    System.Diagnostics.EventLogEntryType.Warning);
                
                // 选项1：立即锁屏
                // TriggerLockScreen(LockReason.VirtualMachineDetected);
                
                // 选项2：记录行为并降低时间限制
                // LogVirtualMachineUsage();
            }
        }
    }
}
```

### 5. 配置文件加密 ⭐⭐⭐⭐

**问题**：原方案使用明文JSON，孩子可修改配置

**解决方案**：加密配置文件

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using System.Text.Json;

namespace ChildControl.Protection
{
    public class SecureConfigStorage
    {
        private static readonly byte[] EncryptionKey = 
            Encoding.UTF8.GetBytes("Your256BitKeyHere!!ChangeThis!!");
        private static readonly byte[] IV = 
            Encoding.UTF8.GetBytes("16BytesVector123");
        
        /// <summary>
        /// 加密并保存配置
        /// </summary>
        public static void SaveConfig<T>(T config, string filePath)
        {
            try
            {
                var json = JsonSerializer.Serialize(config);
                var encrypted = Encrypt(json);
                File.WriteAllBytes(filePath, encrypted);
            }
            catch (Exception ex)
            {
                EventLog.WriteEntry($"SaveConfig error: {ex.Message}", 
                    System.Diagnostics.EventLogEntryType.Error);
            }
        }
        
        /// <summary>
        /// 解密并加载配置
        /// </summary>
        public static T LoadConfig<T>(string filePath)
        {
            try
            {
                if (!File.Exists(filePath))
                    return default(T);
                
                var encrypted = File.ReadAllBytes(filePath);
                var json = Decrypt(encrypted);
                return JsonSerializer.Deserialize<T>(json);
            }
            catch
            {
                return default(T);
            }
        }
        
        private static string Encrypt(string plainText)
        {
            using (var aes = Aes.Create())
            {
                aes.Key = EncryptionKey;
                aes.IV = IV;
                
                using (var encryptor = aes.CreateEncryptor(aes.Key, aes.IV))
                using (var msEncrypt = new MemoryStream())
                {
                    using (var csEncrypt = new CryptoStream(msEncrypt, encryptor, CryptoStreamMode.Write))
                    using (var swEncrypt = new StreamWriter(csEncrypt))
                    {
                        swEncrypt.Write(plainText);
                    }
                    return Convert.ToBase64String(msEncrypt.ToArray());
                }
            }
        }
        
        private static string Decrypt(string cipherText)
        {
            var cipherBytes = Convert.FromBase64String(cipherText);
            using (var aes = Aes.Create())
            {
                aes.Key = EncryptionKey;
                aes.IV = IV;
                
                using (var decryptor = aes.CreateDecryptor(aes.Key, aes.IV))
                using (var msDecrypt = new MemoryStream(cipherBytes))
                using (var csDecrypt = new CryptoStream(msDecrypt, decryptor, CryptoStreamMode.Read))
                using (var srDecrypt = new StreamReader(csDecrypt))
                {
                    return srDecrypt.ReadToEnd();
                }
            }
        }
    }
}
```

### 6. 安全模式防护 ⭐⭐⭐

**问题**：孩子可能进入安全模式绕过程序

**解决方案**：检测安全模式并阻止

```csharp
using System;
using Microsoft.Win32;

namespace ChildControl.Protection
{
    public class SafeModeDetector
    {
        [DllImport("kernel32.dll")]
        private static extern bool GetSystemMetrics(int nIndex);
        
        private const int SM_CLEANBOOT = 67;
        
        /// <summary>
        /// 检测是否在安全模式
        /// </summary>
        public static bool IsSafeMode()
        {
            try
            {
                int cleanBoot = GetSystemMetrics(SM_CLEANBOOT);
                return cleanBoot != 0; // 0=正常, 1=带网络, 2=无网络
            }
            catch
            {
                return false;
            }
        }
        
        /// <summary>
        /// 检查安全模式注册表
        /// </summary>
        public static bool IsSafeModeViaRegistry()
        {
            try
            {
                using (var key = Registry.LocalMachine.OpenSubKey(@"SYSTEM\CurrentControlSet\Control\SafeBoot"))
                {
                    return key != null;
                }
            }
            catch
            {
                return false;
            }
        }
        
        /// <summary>
        /// 处理安全模式检测
        /// </summary>
        public static void HandleSafeMode()
        {
            if (IsSafeMode() || IsSafeModeViaRegistry())
            {
                // 记录安全模式进入行为
                LogSafeModeEntry();
                
                // 关机或重启
                ShutdownImmediately();
            }
        }
        
        private static void LogSafeModeEntry()
        {
            try
            {
                string logEntry = $"[{DateTime.Now}] Safe mode detected!";
                File.AppendAllText(@"C:\ProgramData\ChildControl\logs\security.log", 
                    logEntry + Environment.NewLine);
            }
            catch { }
        }
        
        private static void ShutdownImmediately()
        {
            var psi = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "shutdown",
                Arguments = "/s /f /t 0",
                UseShellExecute = false,
                CreateNoWindow = true
            };
            System.Diagnostics.Process.Start(psi);
        }
    }
}
```

### 7. PE启动防护（配合BitLocker）⭐⭐⭐⭐⭐

**问题**：孩子可能用PE系统绕过限制

**解决方案**：检测PE环境 + BitLocker验证

```csharp
using System;
using Microsoft.Win32;
using System.IO;

namespace ChildControl.Protection
{
    public class PEDetector
    {
        /// <summary>
        /// 检测是否在PE系统
        /// </summary>
        public static bool IsPEEnvironment()
        {
            return CheckWinPERegistry() ||
                   CheckDiskType() ||
                   CheckBootType();
        }
        
        private static bool CheckWinPERegistry()
        {
            try
            {
                using (var key = Registry.LocalMachine.OpenSubKey(@"SYSTEM\CurrentControlSet\Control\MiniNT"))
                {
                    return key != null; // PE环境会创建此键
                }
            }
            catch { }
            return false;
        }
        
        private static bool CheckDiskType()
        {
            try
            {
                // PE系统通常从RAM运行
                var systemDrive = Environment.GetFolderPath(Environment.SpecialFolder.System);
                var driveInfo = new DriveInfo(systemDrive);
                return driveInfo.DriveType == DriveType.Ram;
            }
            catch { }
            return false;
        }
        
        private static bool CheckBootType()
        {
            try
            {
                using (var key = Registry.LocalMachine.OpenSubKey(@"SYSTEM\Setup"))
                {
                    if (key != null)
                    {
                        var setupType = key.GetValue("SetupType")?.ToString();
                        return setupType == "1"; // PE系统SetupType=1
                    }
                }
            }
            catch { }
            return false;
        }
        
        /// <summary>
        /// 检查BitLocker状态
        /// </summary>
        public static bool IsBitLockerEnabled()
        {
            try
            {
                using (var key = Registry.LocalMachine.OpenSubKey(
                    @"SOFTWARE\Microsoft\Windows\CurrentVersion\BitLocker"))
                {
                    if (key != null)
                    {
                        var status = key.GetValue("VolumeStatus")?.ToString();
                        return status == "1"; // 已加密
                    }
                }
            }
            catch { }
            return false;
        }
        
        /// <summary>
        /// 处理PE环境
        /// </summary>
        public static void HandlePEEnvironment()
        {
            if (IsPEEnvironment())
            {
                EventLog.WriteEntry("PE environment detected!", 
                    System.Diagnostics.EventLogEntryType.Warning);
                
                // PE环境下立即关机
                ShutdownImmediately();
            }
        }
        
        private static void ShutdownImmediately()
        {
            var psi = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "shutdown",
                Arguments = "/s /f /t 0",
                UseShellExecute = false,
                CreateNoWindow = true
            };
            System.Diagnostics.Process.Start(psi);
        }
    }
}
```
