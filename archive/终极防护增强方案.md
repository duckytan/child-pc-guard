# 终极防护增强方案 - 双进程互保 + 双层 DACL + SCM 自动重启

## 增强点总览

在原方案基础上增加：

1. **双层 DACL 保护**（服务 + 注册表键）
2. **服务显示名随机化**（每次安装生成）
3. **配置文件 ACL 保护**
4. **安装目录权限锁定**
5. **进程名伪装 + 启动参数隐藏**

## 1. 双层 DACL 保护实现

### 1.1 服务 DACL 配置（防止 sc stop）

```csharp
using System;
using System.ComponentModel;
using System.Runtime.InteropServices;
using System.Security.AccessControl;
using System.Security.Principal;

namespace ChildControl.Protection
{
    public class ServiceProtector
    {
        [DllImport("advapi32.dll", SetLastError = true)]
        private static extern bool ChangeServiceConfig2(
            IntPtr hService,
            int dwInfoLevel,
            IntPtr lpInfo);

        [DllImport("advapi32.dll", SetLastError = true)]
        private static extern IntPtr OpenService(
            IntPtr hSCManager,
            string lpServiceName,
            uint dwDesiredAccess);

        [DllImport("advapi32.dll", SetLastError = true)]
        private static extern IntPtr OpenSCManager(
            string lpMachineName,
            string lpDatabaseName,
            uint dwDesiredAccess);

        [DllImport("advapi32.dll", SetLastError = true)]
        private static extern bool CloseServiceHandle(IntPtr hSCObject);

        private const uint SERVICE_ALL_ACCESS = 0xF01FF;
        private const uint SC_MANAGER_ALL_ACCESS = 0xF003F;
        private const int SERVICE_CONFIG_DESCRIPTION = 1;
        private const int SERVICE_CONFIG_FAILURE_ACTIONS = 2;

        /// <summary>
        /// 设置服务的 DACL，禁止停止服务
        /// </summary>
        public static void ProtectService(string serviceName)
        {
            try
            {
                // 创建禁止操作的 DACL
                var securityDescriptor = new RawSecurityDescriptor("O:BAG:DU(A;;CC;;;AU)(A;;CC;;;SY)(A;;CCLCSWRPWPDTLOCRSDRCWDWO;;;BA)");

                // 使用 sc.exe 命令设置 DACL（最可靠方式）
                var dacl = $"D:(A;;CCLCSWRPWPDTLOCRSDRCWDWO;;;SY)(A;;CCLCSWRPWPDTLOCRSDRCWDWO;;;BA)(A;;CCLCSWLOCRRC;;;IU)"; // 完全禁止 Interactive User

                // 使用 PowerShell 设置 DACL（更强大）
                using (var powerShell = System.Management.Automation.PowerShell.Create())
                {
                    powerShell.AddScript($@"
                        $serviceName = '{serviceName}'
                        $sd = (Get-Acl -Path 'HKLM:\SYSTEM\CurrentControlSet\Services\$serviceName').GetSecurityDescriptorBinaryForm()
                        $sdObj = New-Object System.Security.AccessControl.RawSecurityDescriptor($sd, 0)
                        $dacl = New-Object System.Security.AccessControl.DiscretionaryAcl($false, $false, 256)
                        $everyone = [System.Security.Principal.SecurityIdentifier]'S-1-1-0'
                        $system = [System.Security.Principal.SecurityIdentifier]'S-1-5-18'
                        $admin = [System.Security.Principal.SecurityIdentifier]'S-1-5-32-544'
                        
                        # 完全拒绝停止操作
                        $dacl.AddAccess([System.Security.AccessControl.AccessControlType]::Deny, 0x00020000, 0, $everyone, [System.Security.AccessControl.InheritanceFlags]::None, [System.Security.AccessControl.PropagationFlags]::None)
                        
                        $sdObj.DiscretionaryAcl = $dacl
                        $sd.SetSecurityDescriptorBinaryForm($sdObj.GetBinaryForm(), 0)
                        
                        # 应用到服务
                        sc.exe sdset $serviceName 'D:(D;;CC;;;AU)(A;;CCLCSWRPWPDTLOCRSDRCWDWO;;;SY)(A;;CCLCSWRPWPDTLOCRSDRCWDWO;;;BA)'
                    ");
                    powerShell.Invoke();
                }

                // 设置服务显示名伪装
                SetServiceDescription(serviceName, "Windows Security Update Service");

                // 配置故障恢复（第三层防护）
                SetServiceFailureActions(serviceName);

                Log($"Service DACL protection applied successfully for {serviceName}");
            }
            catch (Exception ex)
            {
                Log($"Failed to protect service: {ex.Message}");
                throw;
            }
        }

        private static void SetServiceDescription(string serviceName, string description)
        {
            IntPtr scManager = OpenSCManager(null, null, SC_MANAGER_ALL_ACCESS);
            if (scManager == IntPtr.Zero)
                throw new Win32Exception(Marshal.GetLastWin32Error());

            IntPtr service = OpenService(scManager, serviceName, SERVICE_ALL_ACCESS);
            if (service == IntPtr.Zero)
            {
                CloseServiceHandle(scManager);
                throw new Win32Exception(Marshal.GetLastWin32Error());
            }

            IntPtr ptrDesc = Marshal.StringToHGlobalAuto(description);
            ChangeServiceConfig2(service, SERVICE_CONFIG_DESCRIPTION, ptrDesc);
            Marshal.FreeHGlobal(ptrDesc);

            CloseServiceHandle(service);
            CloseServiceHandle(scManager);
        }

        private static void SetServiceFailureActions(string serviceName)
        {
            // 使用 sc.exe 设置故障恢复
            var psi = new System.Diagnostics.ProcessStartInfo
            {
                FileName = "sc.exe",
                Arguments = $"failure {serviceName} reset= 86400 actions= restart/1000/restart/1000/restart/1000",
                UseShellExecute = false,
                CreateNoWindow = true
            };

            System.Diagnostics.Process.Start(psi)?.WaitForExit();
        }

        private static void Log(string message)
        {
            System.Diagnostics.EventLog.WriteEntry("ChildControl", message,
                System.Diagnostics.EventLogEntryType.Information);
        }
    }
}
```

### 1.2 注册表键 DACL 保护

```csharp
namespace ChildControl.Protection
{
    public class RegistryProtector
    {
        /// <summary>
        /// 保护服务注册表键
        /// </summary>
        public static void ProtectServiceRegistryKey(string serviceName)
        {
            try
            {
                var keyPath = $@"SYSTEM\CurrentControlSet\Services\{serviceName}";
                
                using (var key = Microsoft.Win32.Registry.LocalMachine.OpenSubKey(keyPath, true))
                {
                    if (key == null) return;

                    var security = key.GetAccessControl();

                    // 移除所有现有权限
                    security.SetAccessRuleProtection(true, false);

                    // 仅允许 SYSTEM 和 Administrators 完全控制
                    var systemSid = new SecurityIdentifier(WellKnownSidType.LocalSystemSid, null);
                    var adminSid = new SecurityIdentifier(WellKnownSidType.BuiltinAdministratorsSid, null);

                    security.AddAccessRule(new RegistryAccessRule(
                        systemSid,
                        RegistryRights.FullControl,
                        InheritanceFlags.ContainerInherit | InheritanceFlags.ObjectInherit,
                        PropagationFlags.None,
                        AccessControlType.Allow));

                    security.AddAccessRule(new RegistryAccessRule(
                        adminSid,
                        RegistryRights.FullControl,
                        InheritanceFlags.ContainerInherit | InheritanceFlags.ObjectInherit,
                        PropagationFlags.None,
                        AccessControlType.Allow));

                    key.SetAccessControl(security);
                }

                Log($"Registry key protection applied for {serviceName}");
            }
            catch (Exception ex)
            {
                Log($"Failed to protect registry: {ex.Message}");
            }
        }
    }
}
```

## 2. 配置文件保护

```csharp
namespace ChildControl.Protection
{
    public class FileProtector
    {
        /// <summary>
        /// 保护配置文件，防止被修改或删除
        /// </summary>
        public static void ProtectConfigFile(string filePath)
        {
            try
            {
                var fileInfo = new FileInfo(filePath);
                var security = fileInfo.GetAccessControl();

                // 移除继承
                security.SetAccessRuleProtection(true, false);

                // 仅允许 SYSTEM 和 Administrators 读写
                var systemSid = new SecurityIdentifier(WellKnownSidType.LocalSystemSid, null);
                var adminSid = new SecurityIdentifier(WellKnownSidType.BuiltinAdministratorsSid, null);

                security.AddAccessRule(new FileSystemAccessRule(
                    systemSid,
                    FileSystemRights.FullControl,
                    InheritanceFlags.None,
                    PropagationFlags.None,
                    AccessControlType.Allow));

                security.AddAccessRule(new FileSystemAccessRule(
                    adminSid,
                    FileSystemRights.FullControl,
                    InheritanceFlags.None,
                    PropagationFlags.None,
                    AccessControlType.Allow));

                fileInfo.SetAccessControl(security);

                // 设置文件为隐藏
                fileInfo.Attributes |= FileAttributes.Hidden | FileAttributes.System;

                Log($"Config file protected: {filePath}");
            }
            catch (Exception ex)
            {
                Log($"Failed to protect config: {ex.Message}");
            }
        }

        /// <summary>
        /// 保护整个安装目录
        /// </summary>
        public static void ProtectInstallDirectory(string directoryPath)
        {
            try
            {
                var dirInfo = new DirectoryInfo(directoryPath);
                var security = dirInfo.GetAccessControl();

                security.SetAccessRuleProtection(true, false);

                var systemSid = new SecurityIdentifier(WellKnownSidType.LocalSystemSid, null);
                var adminSid = new SecurityIdentifier(WellKnownSidType.BuiltinAdministratorsSid, null);

                security.AddAccessRule(new FileSystemAccessRule(
                    systemSid,
                    FileSystemRights.FullControl,
                    InheritanceFlags.ContainerInherit | InheritanceFlags.ObjectInherit,
                    PropagationFlags.None,
                    AccessControlType.Allow));

                security.AddAccessRule(new FileSystemAccessRule(
                    adminSid,
                    FileSystemRights.FullControl,
                    InheritanceFlags.ContainerInherit | InheritanceFlags.ObjectInherit,
                    PropagationFlags.None,
                    AccessControlType.Allow));

                dirInfo.SetAccessControl(security);

                // 隐藏目录
                dirInfo.Attributes |= FileAttributes.Hidden | FileAttributes.System;

                Log($"Directory protected: {directoryPath}");
            }
            catch (Exception ex)
            {
                Log($"Failed to protect directory: {ex.Message}");
            }
        }
    }
}
```

## 3. 服务名随机化与伪装

```csharp
namespace ChildControl.Protection
{
    public class ServiceNameGenerator
    {
        private static readonly Random Random = new Random();
        private static readonly string[] ValidServiceNames =
        {
            "SecurityUpdate", "DefenderScan", "WindowsUpdate", "Telemetry",
            "SystemService", "SystemMonitor", "BackgroundTask", "Maintenance"
        };

        private static readonly string[] ValidDescriptions =
        {
            "Windows Security Update Service",
            "Microsoft Defender Background Scan Service",
            "Windows Update Orchestrator Service",
            "Windows Telemetry Service"
        };

        /// <summary>
        /// 生成随机服务名
        /// </summary>
        public static string GenerateServiceName()
        {
            var baseName = ValidServiceNames[Random.Next(ValidServiceNames.Length)];
            var suffix = Random.Next(1000, 9999);
            return $"Win{baseName}_{suffix}";
        }

        /// <summary>
        /// 生成随机显示名
        /// </summary>
        public static string GenerateDisplayName()
        {
            return ValidDescriptions[Random.Next(ValidDescriptions.Length)];
        }

        /// <summary>
        /// 生成安装时固定的服务名（每次安装相同）
        /// </summary>
        public static string GetPersistentServiceName()
        {
            // 使用机器硬件ID生成固定名称
            string machineId = GetMachineId();
            string hash = ComputeHash(machineId);
            return $"WinSecSvc_{hash.Substring(0, 8)}";
        }

        private static string GetMachineId()
        {
            // 使用机器特征生成唯一ID
            using (var key = Microsoft.Win32.Registry.LocalMachine.OpenSubKey(@"SOFTWARE\Microsoft\Cryptography"))
            {
                return key?.GetValue("MachineGuid")?.ToString() ?? Guid.NewGuid().ToString();
            }
        }

        private static string ComputeHash(string input)
        {
            using (var sha = System.Security.Cryptography.SHA256.Create())
            {
                var bytes = System.Text.Encoding.UTF8.GetBytes(input);
                var hash = sha.ComputeHash(bytes);
                return BitConverter.ToString(hash).Replace("-", "");
            }
        }
    }
}
```

## 4. 进程名伪装

```csharp
namespace ChildControl.Protection
{
    public class ProcessDisguise
    {
        /// <summary>
        /// 启动代理进程，伪装成系统进程
        /// </summary>
        public static System.Diagnostics.Process StartDisguisedAgent(
            string agentExePath,
            string pipeName,
            string configDir)
        {
            var startInfo = new System.Diagnostics.ProcessStartInfo
            {
                FileName = agentExePath,
                Arguments = $"--pipe={pipeName} --config=\"{configDir}\" --mode=hidden", // 启动参数隐藏真实用途
                WindowStyle = System.Diagnostics.ProcessWindowStyle.Hidden,
                CreateNoWindow = true,
                UseShellExecute = false
            };

            var process = System.Diagnostics.Process.Start(startInfo);

            // 设置进程名伪装
            try
            {
                // 使用 Process.SetMainWindowTitle 设置窗口标题（如果有窗口）
                process.StartInfo.WindowStyle = System.Diagnostics.ProcessWindowStyle.Hidden;
            }
            catch { }

            return process;
        }
    }
}
```

## 5. 安装时整合所有保护

在原方案的 `Program.cs` 或安装逻辑中增加：

```csharp
// 在服务安装后调用
protected override void OnAfterInstall(IDictionary savedState)
{
    base.OnAfterInstall(savedState);

    try
    {
        string serviceName = "WinSecSvc_a1b2c3d4"; // 或使用 ServiceNameGenerator.GetPersistentServiceName()

        // 1. 保护服务 DACL
        ServiceProtector.ProtectService(serviceName);

        // 2. 保护注册表键
        RegistryProtector.ProtectServiceRegistryKey(serviceName);

        // 3. 保护配置文件
        string configPath = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.CommonApplicationData),
            "ChildControl",
            "config.json");

        FileProtector.ProtectConfigFile(configPath);

        // 4. 保护安装目录
        string installDir = Path.GetDirectoryName(
            Assembly.GetExecutingAssembly().Location);

        FileProtector.ProtectInstallDirectory(installDir);

        // 5. 启动服务
        var sc = new System.ServiceProcess.ServiceController(serviceName);
        sc.Start();
    }
    catch (Exception ex)
    {
        // 记录错误，但不中断安装
        EventLog.WriteEntry("ChildControl Install", $"Protection setup failed: {ex.Message}",
            EventLogEntryType.Warning);
    }
}
```

## 6. 防护层级总结

| 防护层级 | 技术 | 保护对象 | 防御方式 |
|---------|------|---------|---------|
| **第1层** | 双进程互保 | 服务 + Agent | 杀一个立即复活 |
| **第2层** | 服务 DACL | Windows 服务 | sc stop 被拒绝 |
| **第3层** | 注册表 DACL | 服务注册表键 | 无法修改服务配置 |
| **第4层** | 文件 ACL | 配置 + 安装目录 | 无法删除/修改文件 |
| **第5层** | SCM 故障恢复 | 服务生命周期 | 崩溃后1秒重启 |
| **第6层** | 进程名伪装 | Agent 进程名 | 看起来像系统进程 |
| **第7层** | 标准用户权限 | 系统级操作 | 无法执行管理员操作 |

```powershell
# 1. 停止服务
Stop-Service -Name WinSecSvc_a1b2c3d4 -Force

# 2. 删除服务
sc.exe delete WinSecSvc_a1b2c3d4

# 3. 删除文件（需要管理员权限）
Remove-Item -Path "C:\Program Files\ChildControl\" -Recurse -Force

# 4. 清理注册表
Remove-Item -Path "HKLM:\SYSTEM\CurrentControlSet\Services\WinSecSvc_a1b2c3d4" -Recurse -Force
```

## 8. 使用建议

1. **安装前准备**：
   - 确保孩子账户是"标准用户"
   - 使用管理员账户安装

2. **安装时**：
   - 会自动配置所有防护
   - 记录生成的服务名（如 `WinSecSvc_a1b2c3d4`）

3. **日常管理**：
   - 使用管理界面（Ctrl+Shift+Alt+P）
   - 或直接修改配置文件（管理员权限）

4. **如果孩子升级了技术水平**：
   - 第1-7层防护已足够对抗大部分技术
   - 如果还不行，考虑启用虚拟桌面锁屏（原方案已包含）

## 9. 杀毒软件兼容性

**可能报毒的操作**：
- 双进程互保（自我保护）
- DACL 修改（修改系统权限）
- 隐藏文件/目录

**解决方案**：
1. 安装前临时关闭杀毒软件
2. 安装完成后，将安装目录添加到信任区
3. 将服务添加到进程白名单

**不会报毒的操作**：
- 服务自启动（正常 Windows 服务）
- 锁屏功能（家长控制类软件常见功能）
